<body>
    <div class="schedule-container">
        <div id="schedule" class="schedule">
            <div class="day">Senin</div>
            <div class="subjects">PPKn, MTK</div>
        </div>
    </div>

    <script>
        const schedules = [
            { day: "Senin", subjects: "PPKn, MTK" },
            { day: "Selasa", subjects: "Kimia, B. Inggris, PKWU" },
            { day: "Rabu", subjects: "Informatika, B. Indo, Ekonomi" },
            { day: "Kamis", subjects: "Pend Agama, Seni Budaya, Geografi" },
            { day: "Jumat", subjects: "Sejarah Indo, Pramuka (TLS)" },
            { day: "Senin (2)", subjects: "PJOK, Pramuka (PRK)" }
        ];

        let index = 0;
        const scheduleElement = document.getElementById('schedule');

        function updateSchedule() {
            scheduleElement.innerHTML = `
                <div class="day">${schedules[index].day}</div>
                <div class="subjects">${schedules[index].subjects}</div>
            `;
            scheduleElement.classList.remove('schedule');
            void scheduleElement.offsetWidth; // Trigger reflow to restart animation
            scheduleElement.classList.add('schedule');

            index = (index + 1) % schedules.length;
        }

        setInterval(updateSchedule, 3000); // Update every 3 seconds
    </script>
